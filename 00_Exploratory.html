COMP = rf.benchmark %>% 
  tbl_df %>% 
  rename(benchmark=cost) %>% 
  mutate(
    sub_rf_2015718 = read.csv("reports/submission_rf_20150718_RF-3-ge16d7f8-dirty.csv")[,2]
  ) %>% 
  mutate_each( funs( benchmark - . ), starts_with("sub_")) %>% 
  gather( metric, value, -id, -benchmark )
